# herrapack
script de  instalacion de herramientas basicas
para termux.

comandos de instalacion.

$ termux-setup-storage

$ apt update && apt upgrade -y

$ pkg install git -y

$ git clone https://github.com/capitancomando/herrapack.sh.git

$ ls

$ cd herrapack.sh

$ chomod 711 herrapack.sh

$ ls

$ bash herrapack.sh
